    <script>
        const { createApp } = Vue;

        createApp({
            data() {
                return {
                    currentTheme: '',
                    currentTime: '',
                    sidebarCollapsed: false,
                    currentPage: 'home',
                    currentAppUrl: '',
                    alertPanelCollapsedUp: false,
                    alertPanelCollapsedLeft: false,
                    serverAlerts: [],
                    processAlerts: [],
                    clearedAlerts: new Set(),
                    lastUpdated: '',
                    applications: [
                        {
                            id: 'monitoring_server',
                            name: 'Monitoring Server',
                            icon: 'üñ•Ô∏è',
                            url: 'http://192.168.2.137:8080/monitoring_server/home'
                        },
                        {
                            id: 'monitoring_datasets',
                            name: 'Monitoring Datasets',
                            icon: 'üìä',
                            url: 'http://192.168.2.137:8080/monitoring_datasets/'
                        },
                        {
                            id: 'process_monitoring',
                            name: 'Process Monitoring',
                            icon: '‚öôÔ∏è',
                            url: 'http://192.168.2.137:8080/process_monitoring/'
                        },
                        {
                            id: 'ridam_registration',
                            name: 'RIDAM Registration',
                            icon: 'üìù',
                            url: 'http://192.168.2.137:8080/ridam_registration/'
                        },
                        {
                            id: 'monitoring_sentinel',
                            name: 'Monitoring Sentinel',
                            icon: 'üõ∞Ô∏è',
                            url: 'http://192.168.2.137:8080/monitoring_sentinel/'
                        }
                    ]
                };
            },
            computed: {
                totalAlerts() {
                    return this.serverAlerts.length + this.processAlerts.length;
                }
            },
            methods: {
                toggleSidebar() {
                    this.sidebarCollapsed = !this.sidebarCollapsed;
                },
                navigateTo(page) {
                    this.currentPage = page;
                    if (page === 'home') {
                        this.currentAppUrl = '';
                    }
                },
                setActive(appId) {
                    this.currentPage = appId;
                    const app = this.applications.find(a => a.id === appId);
                    if (app) {
                        this.currentAppUrl = app.url;
                    }
                },
                toggleAlertPanelUp() {
                    this.alertPanelCollapsedUp = !this.alertPanelCollapsedUp;
                },
                toggleAlertPanelLeft() {
                    this.alertPanelCollapsedLeft = !this.alertPanelCollapsedLeft;
                },
                updateTime() {
                    const now = new Date();
                    const options = { 
                        weekday: 'short', 
                        year: 'numeric', 
                        month: 'short', 
                        day: 'numeric',
                        hour: '2-digit',
                        minute: '2-digit',
                        second: '2-digit'
                    };
                    this.currentTime = now.toLocaleString('en-IN', options);
                },
                async fetchAlerts() {
                    try {
                        // Fetch server alerts
                        const serverResponse = await fetch('http://192.168.2.137:8080/monitoring_server/api/alerts');
                        const serverData = await serverResponse.json();

                        // Fetch process alerts
                        const processResponse = await fetch('http://192.168.2.137:8080/process_monitoring/get-alerts');
                        const processData = await processResponse.json();

                        // Filter out cleared alerts and update
                        if (serverData.alerts) {
                            this.serverAlerts = serverData.alerts.filter(alert => 
                                !this.clearedAlerts.has(this.getAlertKey('server', alert))
                            );
                        }

                        if (processData.data && processData.data.alerts) {
                            this.processAlerts = processData.data.alerts.filter(alert => 
                                !this.clearedAlerts.has(this.getAlertKey('process', alert))
                            );
                            this.lastUpdated = processData.last_updated_time || '';
                        }

                        // Check if any cleared alerts have reappeared
                        const currentAlertKeys = new Set();
                        [...serverData.alerts || [], ...(processData.data?.alerts || [])].forEach(alert => {
                            const key = this.getAlertKey('server', alert) || this.getAlertKey('process', alert);
                            currentAlertKeys.add(key);
                        });

                        // Remove cleared alerts that no longer exist from the cleared set
                        this.clearedAlerts = new Set(
                            [...this.clearedAlerts].filter(key => currentAlertKeys.has(key))
                        );

                    } catch (error) {
                        console.error('Error fetching alerts:', error);
                    }
                },
                getAlertKey(type, alert) {
                    return `${type}-${alert.server}-${alert.type}-${alert.message}`;
                },
                removeAlert(type, index) {
                    if (type === 'server') {
                        const alert = this.serverAlerts[index];
                        this.clearedAlerts.add(this.getAlertKey('server', alert));
                        this.serverAlerts.splice(index, 1);
                    } else {
                        const alert = this.processAlerts[index];
                        this.clearedAlerts.add(this.getAlertKey('process', alert));
                        this.processAlerts.splice(index, 1);
                    }
                },
                clearAllAlerts() {
                    this.serverAlerts.forEach(alert => {
                        this.clearedAlerts.add(this.getAlertKey('server', alert));
                    });
                    this.processAlerts.forEach(alert => {
                        this.clearedAlerts.add(this.getAlertKey('process', alert));
                    });
                    this.serverAlerts = [];
                    this.processAlerts = [];
                }
            },
            watch: {
                currentTheme(newTheme) {
                    document.body.className = newTheme;
                }
            },
            mounted() {
                // Update time immediately and every second
                this.updateTime();
                setInterval(this.updateTime, 1000);

                // Fetch alerts immediately and every minute
                this.fetchAlerts();
                setInterval(this.fetchAlerts, 60000);
            }
        }).mount('#app');
    </script>
