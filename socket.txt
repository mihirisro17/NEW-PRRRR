    <script src="https://cdn.socket.io/4.5.4/socket.io.min.js"></script>
    <script>
        // ==========================================
        // STATE MANAGEMENT
        // ==========================================
        const state = {
            serverAlerts: [],
            datasetAlerts: [],
            processAlerts: [],
            storageAlerts: [],
            alertHistory: [],
            clearedAlerts: new Set(),
            lastUpdated: '',
            socket: null,
            wsConnected: false
        };

        // ==========================================
        // HELPER FUNCTIONS
        // ==========================================
        function getAlertKey(category, alert) {
            return `${category}|${alert.type}|${alert.message}|${alert.server}`;
        }

        function formatTime(date = new Date()) {
            return date.toLocaleString('en-IN', {
                month: 'short',
                day: 'numeric',
                hour: '2-digit',
                minute: '2-digit',
                second: '2-digit'
            });
        }

        function scrollToAlerts() {
            document.getElementById('alert-details').scrollIntoView({ behavior: 'smooth' });
        }

        // ==========================================
        // WEBSOCKET CONNECTION
        // ==========================================
        function initWebSocket() {
            const serverUrl = 'http://192.168.2.137:8080'; // Change to your server IP if needed

            console.log('üîå Connecting to WebSocket server:', serverUrl);

            try {
                state.socket = io(serverUrl, {
                    transports: ['websocket', 'polling'],
                    reconnection: true,
                    reconnectionDelay: 1000,
                    reconnectionAttempts: 10
                });

                state.socket.on('connect', () => {
                    console.log('‚úÖ WebSocket connected');
                    state.wsConnected = true;
                    updateConnectionStatus(true);
                });

                state.socket.on('disconnect', () => {
                    console.log('üîå WebSocket disconnected');
                    state.wsConnected = false;
                    updateConnectionStatus(false);
                });

                state.socket.on('message', (data) => {
                    try {
                        const notification = typeof data === 'string' ? JSON.parse(data) : data;
                        console.log('üì© Received alert:', notification);
                        handleIncomingAlert(notification);
                    } catch (e) {
                        console.error('Error parsing message:', e);
                    }
                });

                state.socket.on('connect_error', (error) => {
                    console.error('‚ùå Connection error:', error);
                    updateConnectionStatus(false);
                });

            } catch (error) {
                console.error('Failed to initialize WebSocket:', error);
                updateConnectionStatus(false);
            }
        }

        function updateConnectionStatus(connected) {
            const statusEl = document.getElementById('connection-status');
            if (connected) {
                statusEl.className = 'connection-status connected';
                statusEl.textContent = '‚úÖ Connected';
            } else {
                statusEl.className = 'connection-status disconnected';
                statusEl.textContent = '‚ö†Ô∏è Disconnected';
            }
        }

        // ==========================================
        // ALERT HANDLING
        // ==========================================
        function handleIncomingAlert(notification) {
            if (!notification.category || !notification.type || !notification.message) {
                console.warn('Invalid notification format:', notification);
                return;
            }

            const category = notification.category.toLowerCase();
            const alert = {
                type: notification.type,
                message: notification.message,
                server: notification.server || 'Unknown',
                timestamp: notification.timestamp || new Date().toISOString(),
                metadata: notification.metadata || {}
            };

            const key = getAlertKey(category, alert);
            if (state.clearedAlerts.has(key)) {
                console.log('Alert was previously cleared, ignoring:', key);
                return;
            }

            const categoryMap = {
                server: state.serverAlerts,
                dataset: state.datasetAlerts,
                process: state.processAlerts,
                storage: state.storageAlerts
            };

            const arr = categoryMap[category];
            if (!arr) {
                console.warn('Unknown category:', category);
                return;
            }

            const exists = arr.some(a =>
                a.type === alert.type &&
                a.message === alert.message &&
                a.server === alert.server
            );

            if (!exists) {
                arr.unshift(alert);
                state.lastUpdated = formatTime();
                console.log(`Added ${category} alert:`, alert);
                renderAll();
            }
        }

        function removeAlert(category, index) {
            const categoryMap = {
                server: state.serverAlerts,
                dataset: state.datasetAlerts,
                process: state.processAlerts,
                storage: state.storageAlerts
            };

            const arr = categoryMap[category];
            if (!arr || index < 0 || index >= arr.length) return;

            const alert = arr[index];
            arr.splice(index, 1);

            const timeStr = formatTime();
            state.alertHistory.unshift({ ...alert, category, clearedAt: timeStr });
            const key = getAlertKey(category, alert);
            state.clearedAlerts.add(key);

            if (state.alertHistory.length > 50) {
                state.alertHistory = state.alertHistory.slice(0, 50);
            }

            // Notify server (optional)
            if (state.socket && state.wsConnected) {
                state.socket.emit('message', JSON.stringify({
                    action: 'clear',
                    category,
                    alert
                }));
            }

            renderAll();
        }

        function clearAllAlerts() {
            if (!confirm('Clear all alerts? This will move them to history.')) return;

            const timeStr = formatTime();
            ['server', 'dataset', 'process', 'storage'].forEach(cat => {
                const categoryMap = {
                    server: state.serverAlerts,
                    dataset: state.datasetAlerts,
                    process: state.processAlerts,
                    storage: state.storageAlerts
                };

                const arr = categoryMap[cat];
                arr.forEach(alert => {
                    state.alertHistory.unshift({ ...alert, category: cat, clearedAt: timeStr });
                    const key = getAlertKey(cat, alert);
                    state.clearedAlerts.add(key);
                });
                arr.length = 0;
            });

            if (state.alertHistory.length > 50) {
                state.alertHistory = state.alertHistory.slice(0, 50);
            }

            renderAll();
        }

        // RENDERING FUNCTIONS

        function createAlertItem(category, alert, index) {
            const div = document.createElement('div');
            const typeClass = `alert-${alert.type.toLowerCase()}`;
            div.className = `alert-item ${typeClass}`;

            div.innerHTML = `
                <div class="alert-content">
                    <div class="alert-type">${alert.type.toUpperCase()}</div>
                    <div class="alert-message">${alert.message}</div>
                    <div class="alert-server">Source: ${alert.server}</div>
                </div>
                <button class="alert-delete-btn" title="Clear alert">&times;</button>
            `;

            div.querySelector('.alert-delete-btn').addEventListener('click', () => {
                removeAlert(category, index);
            });

            return div;
        }

        function renderAlertLists() {
            const map = {
                server: { listId: 'server-alerts-list', arr: state.serverAlerts },
                dataset: { listId: 'dataset-alerts-list', arr: state.datasetAlerts },
                process: { listId: 'process-alerts-list', arr: state.processAlerts },
                storage: { listId: 'storage-alerts-list', arr: state.storageAlerts }
            };

            Object.keys(map).forEach(cat => {
                const info = map[cat];
                const container = document.getElementById(info.listId);
                if (!container) return;

                container.innerHTML = '';
                if (info.arr.length === 0) {
                    const p = document.createElement('div');
                    p.className = 'no-alerts';
                    p.textContent = 'No active alerts üéâ';
                    container.appendChild(p);
                } else {
                    info.arr.forEach((alert, idx) => {
                        container.appendChild(createAlertItem(cat, alert, idx));
                    });
                }
            });

            // Update last updated time
            const lastUpdated = document.getElementById('last-updated-text');
            if (lastUpdated) {
                lastUpdated.textContent = `Last updated: ${state.lastUpdated || 'Never'}`;
            }
        }

        function renderCounts() {
            document.getElementById('count-server').textContent = state.serverAlerts.length;
            document.getElementById('count-dataset').textContent = state.datasetAlerts.length;
            document.getElementById('count-process').textContent = state.processAlerts.length;
            document.getElementById('count-storage').textContent = state.storageAlerts.length;

            document.getElementById('badge-server').textContent = state.serverAlerts.length;
            document.getElementById('badge-dataset').textContent = state.datasetAlerts.length;
            document.getElementById('badge-process').textContent = state.processAlerts.length;
            document.getElementById('badge-storage').textContent = state.storageAlerts.length;

            // Show/hide hints
            ['server', 'dataset', 'process', 'storage'].forEach(cat => {
                const hint = document.getElementById(`hint-${cat}`);
                const count = state[`${cat}Alerts`].length;
                if (hint) {
                    hint.style.display = count > 0 ? 'block' : 'none';
                }
            });
        }

        function renderNotificationIndicator() {
            const total = state.serverAlerts.length +
                state.datasetAlerts.length +
                state.processAlerts.length +
                state.storageAlerts.length;

            const indicator = document.getElementById('notification-indicator');
            const count = document.getElementById('notification-count');

            if (total > 0) {
                indicator.classList.remove('hidden');
                count.textContent = total;
            } else {
                indicator.classList.add('hidden');
            }
        }

        function renderAll() {
            renderCounts();
            renderAlertLists();
            renderNotificationIndicator();
        }

        // ==========================================
        // HISTORY MODAL
        // ==========================================
        function openHistory() {
            const modal = document.getElementById('history-modal');
            const body = document.getElementById('history-body');

            body.innerHTML = '';
            if (state.alertHistory.length === 0) {
                const p = document.createElement('div');
                p.className = 'no-alerts';
                p.textContent = 'No cleared alerts in history';
                body.appendChild(p);
            } else {
                state.alertHistory.forEach(item => {
                    const div = document.createElement('div');
                    div.className = 'history-item';
                    div.innerHTML = `
                        <div class="history-time">‚úÖ Cleared: ${item.clearedAt}</div>
                        <div class="history-message">${item.message}</div>
                        <div class="history-server">Source: ${item.server} | Category: ${item.category}</div>
                    `;
                    body.appendChild(div);
                });
            }

            modal.classList.add('active');
        }

        function closeHistory() {
            document.getElementById('history-modal').classList.remove('active');
        }

        // Close modal on outside click
        document.getElementById('history-modal').addEventListener('click', (e) => {
            if (e.target.id === 'history-modal') {
                closeHistory();
            }
        });

        // ==========================================
        // TIME DISPLAY
        // ==========================================
        function updateTime() {
            const timeEl = document.getElementById('live-time');
            const now = new Date();
            const timeStr = now.toLocaleString('en-IN', {
                hour: '2-digit',
                minute: '2-digit',
                second: '2-digit'
            });
            timeEl.textContent = `‚è∞ ${timeStr}`;
        }

            updateTime();
            setInterval(updateTime, 1000);

            // Initialize WebSocket
            initWebSocket();

            // Initial render
            renderAll();

            console.log('‚úÖ Dashboard ready');
        });

        // Add keyboard shortcut to open history (H key)
        document.addEventListener('keydown', (e) => {
            if (e.key === 'h' || e.key === 'H') {
                openHistory();
            }
            if (e.key === 'Escape') {
                closeHistory();
            }
        });
