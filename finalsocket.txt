mounted() {
    console.log('‚úÖ VEDAS MIS Vue app mounted');
    
    // Set initial theme
    document.body.className = 'theme-vedas';
    
    // Initialize Socket.IO connection
    this.initializeSocket();
    
    // Reset shown break times at midnight
    const now = new Date();
    const midnight = new Date(now);
    midnight.setHours(24, 0, 0, 0);
    const msUntilMidnight = midnight - now;
    setTimeout(() => {
        this.shownBreakTimes.clear();
        setInterval(() => {
            this.shownBreakTimes.clear();
        }, 24 * 60 * 60 * 1000);
    }, msUntilMidnight);
    
    // Update time every second
    this.updateTime();
    setInterval(() => {
        this.updateTime();
    }, 1000);
    
    // ‚úÖ FIX: User activity tracking for auto-change delay
    const resetActivityTimer = () => {
        this.lastUserActivity = Date.now();
        console.log('üëÜ User activity detected, resetting auto-change timer');
    };
    
    // Track mouse and keyboard activity
    document.addEventListener('mousemove', resetActivityTimer);
    document.addEventListener('keydown', resetActivityTimer);
    document.addEventListener('click', resetActivityTimer);
    document.addEventListener('scroll', resetActivityTimer);
    document.addEventListener('touchstart', resetActivityTimer);
    
    // ‚úÖ FIX: Improved auto-change with user activity detection
    const startAutoChange = () => {
        if (this.autoChangeTimer) {
            clearInterval(this.autoChangeTimer);
        }
        
        this.autoChangeTimer = setInterval(() => {
            const timeSinceActivity = Date.now() - this.lastUserActivity;
            
            // If user was active in last 5 minutes, delay the change
            if (timeSinceActivity < this.userActivityDelay) {
                console.log('‚è∏Ô∏è Auto-change delayed due to recent user activity');
                console.log(`‚è∞ Time since last activity: ${Math.floor(timeSinceActivity / 1000)}s`);
                return;
            }
            
            const monitoringApps = this.monitoringApps.filter(app => app.id !== 'monitoringsentinel');
            if (monitoringApps.length === 0) return;
            
            if (this.currentPage !== 'home') {
                let currentIndex = monitoringApps.findIndex(app => app.id === this.currentPage);
                if (currentIndex === -1) {
                    this.setActive(monitoringApps[0].id);
                    console.log(`üîÑ Auto-switched to: ${monitoringApps[0].name}`);
                } else {
                    let nextIndex = (currentIndex + 1) % monitoringApps.length;
                    this.setActive(monitoringApps[nextIndex].id);
                    console.log(`üîÑ Auto-switched to: ${monitoringApps[nextIndex].name}`);
                }
            } else {
                this.setActive(monitoringApps[0].id);
                console.log(`üîÑ Auto-switched from home to: ${monitoringApps[0].name}`);
            }
        }, this.autoChangeDelay); // Check every 10 minutes
    };
    
    startAutoChange();
    
    console.log('‚úÖ VEDAS MIS ready - Listening for alerts via Socket.IO');
    console.log('‚è±Ô∏è Auto-change interval: 10 minutes');
    console.log('‚è±Ô∏è User activity delay: 5 minutes');
}
