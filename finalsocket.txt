    <script>
        const { createApp } = Vue;

        createApp({
            data() {
                return {
                    currentTime: '',
                    currentPage: 'home',
                    currentAppUrl: '',
                    sidebarCollapsed: false,
                    isLoading: false,
                    currentTheme: 'theme-vedas',
                    autoThemeEnabled: false,
                    showHistory: false,
                    collapsedSections: {},

                    alertsByCategory: {
                        server: [],
                        system: []
                    },
                    clearedAlerts: new Set(),
                    processedAlerts: new Set(),
                    alertHistory: [],
                    lastUpdated: '',

                    alertPanelCollapsedUp: false,
                    alertPanelCollapsedLeft: false,
                    serverSectionCollapsed: false,
                    systemSectionCollapsed: false,
                    allSectionsCollapsed: false,

                    totalServers: 20,
                    totalServices: 41,

                    showBreakNotification: false,
                    breakNotification: {},
                    breakAutoCloseSeconds: 150,
                    breakAutoCloseInterval: null,
                    shownBreakTimes: new Set(),

                    socket: null,
                    socketConnected: false,

                    monitoringApps: [
                        { id: 'servermonitor', name: 'Server Monitor', description: 'Infrastructure Status', icon: '🖥️', url: 'http://192.168.2.137:8080/monitoring_server/home' },
                        { id: 'servicemonitor', name: 'Service Monitor', description: 'Data Pipeline Status', icon: '📊', url: 'http://192.168.2.137:8080/monitoring_datasets/' },
                        { id: 'processingscriptmonitor', name: 'Processing Script Monitor', description: 'Service Health Check', icon: '🔄', url: 'http://192.168.2.137:8080/process_monitoring/' },
                        { id: 'datachainmonitoring', name: 'Data Chain Monitoring', description: 'Satellite Tracking', icon: '🛰️', url: 'http://192.168.2.137:8080/monitoring_sentinel/' },
                        { id: 'publishmonitoring', name: 'Publish Monitoring', description: 'Bounds & Search Data', icon: '🔍', url: 'http://192.168.2.137:8080/ridam_dataset_publisher/' },
                        { id: 'ridampoolanalytics', name: 'Ridam Pool Analytics', description: 'Bounds & Search Data', icon: '🗄️', url: 'http://192.168.2.137:8080/pool_storage_monitoring/' },
                        { id: 'poolgrouphealthmonitoring', name: 'Pool Group Health Monitoring', description: 'Bounds & Search Data', icon: '🗃️', url: 'http://192.168.2.137:8080/monitoring_health/' },
                        { id: 'geoentitymonitoring', name: 'Geoentity Monitoring', description: 'Bounds & Search Data', icon: '🌍', url: 'http://192.168.2.137:8080/geoentity_monitoring/' },
                    ],

                    analyticsApps: [
                        { id: 'teamanalytics', name: 'Team Analytics', description: 'Performance Metrics', icon: '📈', url: 'http://192.168.2.137:8080/taskboard/dashboard/' }
                    ],

                    toolsApps: [
                        { id: 'ridam Registration', name: 'Ridam Registration', description: 'Bounds & Search Data', icon: '📝', url: 'http://192.168.2.137:8080/ridam_registration/' }
                    ]
                };
            },

            computed: {
                allApplications() {
                    return [...this.monitoringApps, ...this.analyticsApps, ...this.toolsApps];
                },

                totalAlerts() {
                    return Object.values(this.alertsByCategory).reduce((sum, alerts) => sum + alerts.length, 0);
                }
            },

            methods: {
                toggleSidebar() {
                    this.sidebarCollapsed = !this.sidebarCollapsed;
                },

                setActive(appId) {
                    this.isLoading = true;
                    this.currentPage = appId;

                    if (appId === 'home') {
                        this.currentAppUrl = '';
                        this.isLoading = false;
                        return;
                    }

                    const app = this.allApplications.find(a => a.id === appId);
                    if (app) {
                        this.currentAppUrl = app.url;
                    }
                },

                onIframeLoad() {
                    setTimeout(() => {
                        this.isLoading = false;
                    }, 500);
                },

                handleThemeChange() {
                    if (this.currentTheme === 'auto') {
                        this.autoThemeEnabled = true;
                        this.applyAutoTheme();
                    } else {
                        this.autoThemeEnabled = false;
                        document.body.className = this.currentTheme;
                    }
                },

                applyAutoTheme() {
                    const hour = new Date().getHours();
                    let themeClass;
                    if (hour >= 5 && hour < 12) {
                        themeClass = 'theme-auto-morning';
                    } else if (hour >= 12 && hour < 17) {
                        themeClass = 'theme-auto-afternoon';
                    } else if (hour >= 17 && hour < 21) {
                        themeClass = 'theme-auto-evening';
                    } else {
                        themeClass = 'theme-auto-night';
                    }
                    document.body.className = themeClass;
                },

                updateTime() {
                    const now = new Date();
                    const options = { weekday: 'short', month: 'short', day: 'numeric', hour: '2-digit', minute: '2-digit', second: '2-digit' };
                    this.currentTime = now.toLocaleString('en-IN', options);

                    if (this.autoThemeEnabled) {
                        this.applyAutoTheme();
                    }

                    if (now.getSeconds() === 0) {
                        this.checkBreakTime();
                    }
                },

                checkBreakTime() {
                    const now = new Date();
                    const hours = now.getHours();
                    const minutes = now.getMinutes();
                    const timeKey = `${hours}:${minutes}`;

                    if (this.shownBreakTimes.has(timeKey)) return;

                    const breakTimes = {
                        '11:0': { icon: '☕', title: 'Tea Break!', message: 'Time for a refreshing cup of tea!' },
                        '13:0': { icon: '🍽️', title: 'Lunch Time!', message: 'Enjoy your meal and recharge!' },
                        '16:30': { icon: '☕', title: 'Evening Snacks!', message: 'Take a break and have some snacks!' },
                        '18:30': { icon: '🌆', title: 'Office Time is Over!', message: 'Great work today! Time to head home and relax!' }
                    };

                    const breakInfo = breakTimes[timeKey];
                    if (breakInfo) {
                        this.breakNotification = breakInfo;
                        this.showBreakNotification = true;
                        this.shownBreakTimes.add(timeKey);
                        this.breakAutoCloseSeconds = 150;
                        this.breakAutoCloseInterval = setInterval(() => {
                            this.breakAutoCloseSeconds--;
                            if (this.breakAutoCloseSeconds <= 0) {
                                this.closeBreakNotification();
                            }
                        }, 1000);
                    }
                },

                closeBreakNotification() {
                    this.showBreakNotification = false;
                    if (this.breakAutoCloseInterval) {
                        clearInterval(this.breakAutoCloseInterval);
                        this.breakAutoCloseInterval = null;
                    }
                },

                async manualRefresh() {
                    this.isLoading = true;
                    await this.fetchAlertsFromSocket();
                    setTimeout(() => {
                        this.isLoading = false;
                    }, 1000);
                },

                initializeSocket() {
                    console.log('🔌 Initializing Socket.IO connection...');
                    console.log('📍 Connecting to: http://192.168.2.137:8080');

                    this.socket = io('http://192.168.2.137:8080', {
                        transports: ['websocket', 'polling'],
                        reconnection: true,
                        reconnectionDelay: 1000,
                        reconnectionAttempts: 10,
                        timeout: 20000
                    });

                    // Connection Events
                    this.socket.on('connect', () => {
                        console.log('✅ Socket.IO Connected - ID:', this.socket.id);
                        this.socketConnected = true;
                        this.lastUpdated = `Connected at ${this.formatTime(new Date())}`;
                    });

                    this.socket.on('disconnect', (reason) => {
                        console.log('❌ Socket.IO Disconnected - Reason:', reason);
                        this.socketConnected = false;
                        this.lastUpdated = `Disconnected: ${reason}`;
                    });

                    this.socket.on('connect_error', (error) => {
                        console.error('🔥 Socket.IO Connection Error:', error.message);
                        this.socketConnected = false;
                        this.lastUpdated = 'Connection Error';
                    });

                    // ===== PRIMARY MESSAGE HANDLER (MATCHES YOUR ALERT DASHBOARD) =====
                    this.socket.on('message', (data) => {
                        try {
                            const notification = typeof data === 'string' ? JSON.parse(data) : data;
                            console.log('📩 [MESSAGE EVENT] Received:', notification);
                            this.handleIncomingAlert(notification);
                        } catch (e) {
                            console.error('❌ Error parsing message:', e);
                        }
                    });

                    // Additional event listeners for different event names
                    this.socket.on('alert', (data) => {
                        console.log('🔔 [ALERT EVENT]:', data);
                        this.handleIncomingAlert(data);
                    });

                    this.socket.on('new_alert', (data) => {
                        console.log('🔔 [NEW_ALERT EVENT]:', data);
                        this.handleIncomingAlert(data);
                    });

                    this.socket.on('server_alert', (data) => {
                        console.log('🖥️ [SERVER_ALERT EVENT]:', data);
                        this.handleIncomingAlert({ ...data, category: 'server' });
                    });

                    this.socket.on('dataset_alert', (data) => {
                        console.log('📊 [DATASET_ALERT EVENT]:', data);
                        this.handleIncomingAlert({ ...data, category: 'dataset' });
                    });

                    this.socket.on('process_alert', (data) => {
                        console.log('⚙️ [PROCESS_ALERT EVENT]:', data);
                        this.handleIncomingAlert({ ...data, category: 'process' });
                    });

                    this.socket.on('storage_alert', (data) => {
                        console.log('💾 [STORAGE_ALERT EVENT]:', data);
                        this.handleIncomingAlert({ ...data, category: 'storage' });
                    });

                    // Alert cleared
                    this.socket.on('alert_cleared', (data) => {
                        console.log('✅ [ALERT_CLEARED EVENT]:', data);
                        this.handleAlertCleared(data);
                    });

                    // Clear alerts
                    this.socket.on('clear_alerts', (data) => {
                        console.log('🗑️ [CLEAR_ALERTS EVENT]:', data);
                        if (data.category) {
                            this.clearAlertsByCategory(data.category);
                        } else {
                            this.clearAllAlerts();
                        }
                    });

                    // Stats update
                    this.socket.on('update_stats', (data) => {
                        console.log('📊 [UPDATE_STATS EVENT]:', data);
                        if (data.totalServers !== undefined) this.totalServers = data.totalServers;
                        if (data.totalServices !== undefined) this.totalServices = data.totalServices;
                    });

                    // Category metadata
                    this.socket.on('category_metadata', (data) => {
                        console.log('📋 [CATEGORY_METADATA EVENT]:', data);
                        if (data.category && data.metadata) {
                            this.categoryMetadata[data.category] = data.metadata;
                        }
                    });

                    // Initial alerts load
                    this.socket.on('initial_alerts', (data) => {
                        console.log('📦 [INITIAL_ALERTS EVENT]:', data);
                        Object.keys(data).forEach(category => {
                            this.ensureCategory(category);
                            this.alertsByCategory[category] = data[category] || [];
                        });
                        this.lastUpdated = this.formatTime(new Date());
                    });

                    console.log('✅ All Socket.IO event listeners registered');
                },
                getCategoryIcon(category) {
                    const icons = {
                        'server': '🖥️',
                        'system': '⚠️',
                        'dataset': '📊',
                        'storage': '💾',
                        'network': '🌐',
                        'api': '🔌',
                        'database': '🗄️'
                    };
                    return icons[category] || '🔔';
                },

                getCategoryName(category) {
                    return category.charAt(0).toUpperCase() + category.slice(1) + ' Alerts';
                },

                toggleSection(category) {
                    if (!this.collapsedSections[category]) {
                        this.$set(this.collapsedSections, category, false);
                    }
                    this.collapsedSections[category] = !this.collapsedSections[category];
                },

                // ✅ UPDATE collapseAllSections method:
                collapseAllSections() {
                    this.allSectionsCollapsed = !this.allSectionsCollapsed;
                    Object.keys(this.alertsByCategory).forEach(category => {
                        this.$set(this.collapsedSections, category, this.allSectionsCollapsed);
                    });
                },

                handleIncomingAlert(notification) {
                    console.log('🔔 [handleIncomingAlert] Processing:', notification);

                    if (!notification.category || !notification.type || !notification.message) {
                        console.warn('⚠️ Invalid notification format:', notification);
                        return;
                    }

                    const category = notification.category.toLowerCase();
                    console.log(`📂 Category identified: ${category}`);

                    const alert = {
                        type: notification.type,
                        message: notification.message,
                        server: notification.server || 'Unknown',
                        source: notification.source || category,
                        timestamp: notification.timestamp || new Date().toISOString(),
                        metadata: notification.metadata || {}
                    };

                    const alertKey = this.getAlertKey(category, alert);

                    if (this.processedAlerts.has(alertKey)) {
                        console.log('⏭️ Alert already processed, ignoring duplicate:', alertKey);
                        return;
                    }

                    if (this.clearedAlerts.has(alertKey)) {
                        console.log('⏭️ Alert was previously cleared, ignoring:', alertKey);
                        return;
                    }

                    this.ensureCategory(category);

                    const exists = this.alertsByCategory[category].some(a =>
                        a.type === alert.type &&
                        a.message === alert.message &&
                        a.server === alert.server
                    );

                    if (!exists) {
                        this.alertsByCategory[category].unshift(alert);
                        this.processedAlerts.add(alertKey);
                        console.log(`✅ Alert added to "${category}". Total:`, this.alertsByCategory[category].length);
                    } else {
                        console.log('⚠️ Duplicate alert detected in category, skipping');
                    }

                    this.lastUpdated = this.formatTime(new Date());
                },

                ensureCategory(category) {
                    if (!this.alertsByCategory[category]) {
                        this.alertsByCategory[category] = [];
                    }
                },

                getAlertKey(category, alert) {
                    return `${category}-${alert.server}-${alert.type}-${alert.message}`;
                },

                formatTime(date) {
                    return date.toLocaleString('en-IN', { month: 'short', day: 'numeric', hour: '2-digit', minute: '2-digit' });
                },

                removeAlert(category, index) {
                    if (!this.alertsByCategory[category]) return;

                    const now = new Date();
                    const timeStr = this.formatTime(now);

                    const alert = this.alertsByCategory[category][index];
                    if (!alert) return;

                    this.alertHistory.unshift({ ...alert, category, clearedAt: timeStr });

                    const alertKey = this.getAlertKey(category, alert);
                    this.clearedAlerts.add(alertKey);
                    this.processedAlerts.delete(alertKey);

                    this.alertsByCategory[category].splice(index, 1);

                    if (this.alertHistory.length > 50) {
                        this.alertHistory = this.alertHistory.slice(0, 50);
                    }

                    console.log(`🗑️ Removed alert from "${category}". Remaining:`, this.alertsByCategory[category].length);
                },

                clearAllAlerts() {
                    const now = new Date();
                    const timeStr = this.formatTime(now);

                    Object.keys(this.alertsByCategory).forEach(category => {
                        this.alertsByCategory[category].forEach(alert => {
                            this.alertHistory.unshift({ ...alert, category, clearedAt: timeStr });
                            const alertKey = this.getAlertKey(category, alert);
                            this.clearedAlerts.add(alertKey);
                            this.processedAlerts.delete(alertKey);
                        });
                        this.alertsByCategory[category] = [];
                    });

                    if (this.alertHistory.length > 50) {
                        this.alertHistory = this.alertHistory.slice(0, 50);
                    }

                    console.log('🧹 All alerts cleared');
                },

                // ✅ OPEN ALERT PANEL (WHEN CLICKING BELL)
                openAlertPanel() {
                    this.alertPanelCollapsedLeft = false;
                    this.alertPanelCollapsedUp = false;
                },

                // ✅ CLOSE ALERT PANEL (SLIDES TO RIGHT)
                closeAlertPanel() {
                    this.alertPanelCollapsedUp = false;
                    this.alertPanelCollapsedLeft = true;
                },

                toggleServerSection() {
                    this.serverSectionCollapsed = !this.serverSectionCollapsed;
                },

                toggleSystemSection() {
                    this.systemSectionCollapsed = !this.systemSectionCollapsed;
                },

                // ✅ COLLAPSE/EXPAND ALL SECTIONS
                collapseAllSections() {
                    this.allSectionsCollapsed = !this.allSectionsCollapsed;
                    this.serverSectionCollapsed = this.allSectionsCollapsed;
                    this.systemSectionCollapsed = this.allSectionsCollapsed;
                },

                async fetchAlertsFromSocket() {
                    console.log('🔄 Manual refresh triggered');
                }
            },

            mounted() {
                console.log('🚀 VEDAS MIS Vue app mounted');
                document.body.className = 'theme-vedas';

                this.initializeSocket();

                const now = new Date();
                const midnight = new Date(now);
                midnight.setHours(24, 0, 0, 0);
                const msUntilMidnight = midnight - now;
                setTimeout(() => {
                    this.shownBreakTimes.clear();
                    setInterval(() => this.shownBreakTimes.clear(), 24 * 60 * 60 * 1000);
                }, msUntilMidnight);

                this.updateTime();
                setInterval(this.updateTime, 1000);

                setInterval(() => {
                    const monitoringApps = this.monitoringApps.filter(app => app.id !== 'monitoringsentinel');
                    if (monitoringApps.length === 0) return;

                    if (this.currentPage !== 'home') {
                        let currentIndex = monitoringApps.findIndex(app => app.id === this.currentPage);
                        if (currentIndex === -1) {
                            this.setActive(monitoringApps[0].id);
                        } else {
                            let nextIndex = (currentIndex + 1) % monitoringApps.length;
                            this.setActive(monitoringApps[nextIndex].id);
                        }
                    } else {
                        this.setActive(monitoringApps[0].id);
                    }
                }, 600000);

                console.log('✅ VEDAS MIS ready - Listening for alerts via Socket.IO');
            }
        }).mount('#app');
