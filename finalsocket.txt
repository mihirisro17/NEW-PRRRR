handleIncomingAlert(notification) {
    console.log('üîî [handleIncomingAlert] Processing:', notification);

    if (!notification.category || !notification.type || !notification.message) {
        console.warn('‚ö†Ô∏è Invalid notification format:', notification);
        return;
    }

    const category = notification.category.toLowerCase();
    console.log(`üìÇ Category identified: ${category}`);

    const alert = {
        type: notification.type,
        message: notification.message,
        server: notification.server || 'Unknown',
        source: notification.source || category,
        timestamp: notification.timestamp || new Date().toISOString(),
        metadata: notification.metadata || {}
    };

    // ‚úÖ FIX: Create unique key and check FIRST
    const alertKey = this.getAlertKey(category, alert);

    // Check if already processed OR if already exists in current alerts
    if (this.processedAlerts.has(alertKey)) {
        console.log('‚è≠Ô∏è Alert already processed, ignoring duplicate:', alertKey);
        return;
    }

    if (this.clearedAlerts.has(alertKey)) {
        console.log('‚è≠Ô∏è Alert was previously cleared, ignoring:', alertKey);
        return;
    }

    this.ensureCategory(category);

    // ‚úÖ FIX: More robust duplicate check
    const exists = this.alertsByCategory[category].some(a => {
        const existingKey = this.getAlertKey(category, a);
        return existingKey === alertKey;
    });

    if (!exists) {
        this.alertsByCategory[category].unshift(alert);
        this.processedAlerts.add(alertKey);
        console.log(`‚úÖ Alert added to "${category}". Total:`, this.alertsByCategory[category].length);
    } else {
        console.log('‚ö†Ô∏è Duplicate alert detected in category, skipping');
        this.processedAlerts.add(alertKey); // Mark as processed even if duplicate
    }

    this.lastUpdated = this.formatTime(new Date());
},

mounted() {
    // ... existing mounted code

    // ‚úÖ FIX: User activity tracking
    const resetActivityTimer = () => {
        this.lastUserActivity = Date.now();
        console.log('üëÜ User activity detected, resetting auto-change timer');
    };

    // Track mouse and keyboard activity
    document.addEventListener('mousemove', resetActivityTimer);
    document.addEventListener('keydown', resetActivityTimer);
    document.addEventListener('click', resetActivityTimer);

    // ‚úÖ FIX: Improved auto-change with user activity detection
    const startAutoChange = () => {
        if (this.autoChangeTimer) {
            clearInterval(this.autoChangeTimer);
        }

        this.autoChangeTimer = setInterval(() => {
            const timeSinceActivity = Date.now() - this.lastUserActivity;
            
            // If user was active in last 5 minutes, delay the change
            if (timeSinceActivity < this.userActivityDelay) {
                console.log('‚è∏Ô∏è Auto-change delayed due to recent user activity');
                return;
            }

            const monitoringApps = this.monitoringApps.filter(app => app.id !== 'monitoringsentinel');
            if (monitoringApps.length === 0) return;

            if (this.currentPage !== 'home') {
                let currentIndex = monitoringApps.findIndex(app => app.id === this.currentPage);
                if (currentIndex === -1) {
                    this.setActive(monitoringApps[0].id);
                } else {
                    let nextIndex = (currentIndex + 1) % monitoringApps.length;
                    this.setActive(monitoringApps[nextIndex].id);
                }
            } else {
                this.setActive(monitoringApps[0].id);
            }
        }, this.autoChangeDelay); // Check every 10 minutes
    };

    startAutoChange();

    console.log('‚úÖ VEDAS MIS ready - Listening for alerts via Socket.IO');
}
