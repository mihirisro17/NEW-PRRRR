// ✅ FIX: Remove $set (Vue 3 doesn't need it)
toggleSection(category) {
    // Initialize if doesn't exist
    if (this.collapsedSections[category] === undefined) {
        this.collapsedSections[category] = false;
    }
    // Toggle
    this.collapsedSections[category] = !this.collapsedSections[category];
},

// ✅ FIX: collapseAllSections without $set
collapseAllSections() {
    this.allSectionsCollapsed = !this.allSectionsCollapsed;
    Object.keys(this.alertsByCategory).forEach(category => {
        this.collapsedSections[category] = this.allSectionsCollapsed;
    });
},
