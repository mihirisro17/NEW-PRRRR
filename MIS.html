mounted() { this.updateTime(); setInterval(this.updateTime, 1000); this.fetchAlerts(); setInterval(this.fetchAlerts, 60000); // Switch only MONITORING applications every 10 minutes (exclude sentinel & ridam) setInterval(() => { // Filter only monitoring apps const monitoringApps = this.applications.filter(app => app.id !== 'monitoring_sentinel' && app.id !== 'ridam_registration' ); if (monitoringApps.length === 0) return; if (this.currentPage !== 'home') { // Find current app in monitoring apps only let currentIndex = monitoringApps.findIndex(app => app.id === this.currentPage); // If current app is not in monitoring list, start from first monitoring app if (currentIndex === -1) { this.setActive(monitoringApps[0].id); } else { // Switch to next monitoring app let nextIndex = (currentIndex + 1) % monitoringApps.length; this.setActive(monitoringApps[nextIndex].id); } } else { // If on home page, start with first monitoring app this.setActive(monitoringApps[0].id); } }, 600000); // 10 minutes }
